function animateCounter() {
  let count = 0;
  const target = 100;
  const increment = target / 50;
  
  const timer = setInterval(() => {
    count += increment;
    if (count >= target) {
      count = target;
      clearInterval(timer);
    }
    document.getElementById('counter').textContent = Math.floor(count);
  }, 50);
}
